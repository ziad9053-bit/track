async function addItem(id, price, element) { if (element.dataset.added === "true") return; const orderId = localStorage.getItem("currentOrderId"); const orderCode = localStorage.getItem("currentOrderCode"); if (!orderId || !orderCode) { alert("حدث خطأ: لم يتم إنشاء الطلب بشكل صحيح. يرجى إعادة مسح QR."); return; } await db.from("order_items").insert({ order_id: orderId, item_id: id, quantity: 1, price: price }); element.dataset.added = "true"; const badge = document.createElement("div"); badge.className = "added-badge"; badge.textContent = "✓ تمت الإضافة"; element.appendChild(badge); updateCartBadge(); }
