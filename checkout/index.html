async function confirmOrder() { const orderId = localStorage.getItem("currentOrderId"); const orderCode = localStorage.getItem("currentOrderCode"); if (!orderId) return; // إضافة زر "جاري الإرسال" لتجنب الضغط المتكرر const btn = document.querySelector('button[onclick="confirmOrder()"]'); if (btn) btn.disabled = true; // تغيير حالة الطلب const { error } = await db.from("orders").update({ status: "confirmed" }).eq("id", orderId); if (error) { alert("حدث خطأ، يرجى المحاولة لاحقاً."); if (btn) btn.disabled = false; return; } // رسالة تأكيد لطيفة alert("تم تأكيد طلبك بنجاح! سيتم تحويلك لصفحة التتبع."); // الانتقال لصفحة التتبع window.location.href = `../../customer/track.html?code=${orderCode}`; }
