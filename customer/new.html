// قراءة QR const params = new URLSearchParams(window.location.search); const tableNumFromQR = params.get("table"); // إذا كان هناك QR جديد → احذف الطلب القديم if (tableNumFromQR) { localStorage.removeItem("currentOrderId"); localStorage.removeItem("currentOrderCode"); } else { // لا يوجد QR → المستخدم عاد للصفحة بالخطأ if (localStorage.getItem("currentOrderCode")) { window.location.href = "../menu/index.html?category=hot&code=" + localStorage.getItem("currentOrderCode"); } }
